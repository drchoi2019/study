// 링크 업데이트 함수 - 진짜 자동 감지 방식 function updateAllLinksToParameterStyle(lang) { const linksToUpdate = document.querySelectorAll('a.nav-link'); linksToUpdate.forEach(link => { const originalHref = link.getAttribute('href').split('?')[0]; // 자동 감지: 이미 ?lang= 파라미터가 있으면 파라미터 방식 const isAlreadyParameterBased = link.getAttribute('href').includes('?lang='); if (isAlreadyParameterBased) { // 파라미터 방식 link.href = `${originalHref}?lang=${currentLanguage}`; } else { // _k 방식 const baseHref = originalHref.replace('_k.html', '').replace('.html', ''); if (currentLanguage === 'ko') { link.href = `${baseHref}_k.html`; } else { link.href = `${baseHref}.html`; } } }); }
